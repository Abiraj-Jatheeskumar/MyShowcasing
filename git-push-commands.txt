# Git Commands to Push Project to GitHub
# Copy and paste these commands one by one into your terminal

# 1. Navigate to project directory
cd "C:\Users\aabir\OneDrive\Desktop\Abiraj_Personal"

# 2. Initialize git repository (if not already initialized)
git init

# 3. Add all files to staging
git add .

# 4. Create initial commit
git commit -m "Initial commit: Portfolio website"

# 5. Add remote repository
git remote add origin https://github.com/Abiraj-Jatheeskumar/MyShowcasing.git

# 6. Rename branch to main (if needed)
git branch -M main

# 7. Push to GitHub
git push -u origin main
